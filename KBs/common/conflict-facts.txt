;;;; CONFLICT PREDICATES

; A fight is a conflict. 
(B (name fight) (=> (conflict f :1.2) (fight f)) )

; Combat is a conflict. 
(B (name combat) (=> (conflict c :1.2) (combat c)) )


;;;; CONFLICT ONTOLOGY

;Conflict has two adversaries.
(B (name adversary1) (=> (conflict c :1.2) (adversary e x y c)))

; The adversary relation is symmetric.  *imported from old conflict ontology, may remove*
; (B (name adversary2) (=> (adversary e x y c :1.2)(adversary e1 y x c)) )

;Adversaries have degree of strength.
;   [Let's talk about how to represent this.]

;Adversaries have contradictory goals.
(B (name adversary3) (=> (adversary e x y c :1.2) 
   	 	     	 (^ (conflict c) (goal e1 g x) (goal e2 h y) (not g h))))

;To win is to achieve the goal.
(B (name win) (=> (^ (conflict c) (adversary e x y c) (goal e1 g x) (goal e2 h y) (accomplish-goal e3 g x))
		      (win e x y c)))

;Victory is winning.
(B (name victory) (=> (^ (conflict c :0.3) (adversary e1 x y c :0.3) (win e1 x y c :0.3)) (victory e1 x y c)))

;To lose is for your adversary to win.
(B (name lose) (=> (win e x y c :1.2) (lose e y x c)))

;Adversaries compete with each other.  
(B (name compete) (=> (^ (adversary e1 x y c :0.3) (conflict c :0.6)) (compete e x y u)))


;Adversaries attack each other.
(B (name adversary7) (=> (^ (conflict e :0.6)(adversary e1 x y c :0.6))
   	 	     	 (^ (attack e1 x y)(attack e2 y x))))

;Adversaries use weapons in attacks.
(B (name weapon) (=> (attack e1 x y :1.2) (^ (use e1 x w)(weapon w))) )


;;;; WAR ONTOLOGY

; One kind of conflict is a war. 
(B (name war) (=> (^ (conflict e :0.6)(waretc e :0.6)) (war e)))

;Adversaries in war are groups to which members are loyal.

;A flank is a part of a group adversary that is not strong.

;A frontline is a part of a group adversary that attacks.

;Members of groups sometimes defect.

;Subgroups of groups sometimes revolt.

;Dresden was destroyed by an attack with bombs.
